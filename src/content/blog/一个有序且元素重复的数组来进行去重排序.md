public static int [] get(int [] res){
    if (res.length < 1){
        return res;
    }
    int i = 0, r = i+1;
    while (r < res.length){
        if (res[i]==res[r]){
            r++;
            continue;
        }
        i++;
        res[i]=res[r];
        r++;
    }
    int[] ints = Arrays.copyOf(res, i + 1);
    return ints;
}
